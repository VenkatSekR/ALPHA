DECLARE 
V_COUNTER NUMBER :=0;
V_RESULT NUMBER;
BEGIN
WHILE V_COUNTER<=10 LOOP
V_COUNTER:= V_COUNTER+1;
V_RESULT := V_COUNTER*2;
DBMS_OUTPUT.PUT_LINE(V_COUNTER||'X'||'2'||'='||V_RESULT);
END LOOP;
END;
/

DECLARE 
V_TEST BOOLEAN := TRUE;
V_COUNTER NUMBER :=0;
BEGIN
WHILE V_TEST LOOP
V_COUNTER:= V_COUNTER+1;
DBMS_OUTPUT.PUT_LINE(V_COUNTER);
IF V_COUNTER=100 THEN 
V_TEST := FALSE;
END IF;
END LOOP;
END;
/
