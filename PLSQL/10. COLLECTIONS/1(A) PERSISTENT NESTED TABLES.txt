
1. NESTED TABLES :

SET SERVEROUTPUT ON;
/
DECLARE
  TYPE MY_NESTED_TABLE IS TABLE OF VARCHAR2(15);
  VAR_NT MY_NESTED_TABLE := MY_NESTED_TABLE('A','B','C','D','E','F','G','H');
BEGIN
  FOR I IN 1..VAR_NT.COUNT
  LOOP
  DBMS_OUTPUT.PUT_LINE('THE VALUE STORED IN INDEX'||' '||I||' '||'IS'||' '||VAR_NT(I));
  END LOOP;
END;
/

DECLARE 
TYPE MY_NESTED_TABLE IS TABLE OF NUMBER;
VAR_NT MY_NESTED_TABLE := MY_NESTED_TABLE(1,2,3,4,5,6,7,8,9);
BEGIN
FOR I IN 1..VAR_NT.COUNT
LOOP
DBMS_OUTPUT.PUT_LINE('THE VALUE OF INDEX'||' '||I||' '||'IS'||' '||VAR_NT(I));
END LOOP; 
END;
/
