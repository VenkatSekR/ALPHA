DECLARE 
VV EMP%ROWTYPE;
CURSOR C1 (V1 NUMBER) IS SELECT * FROM EMP WHERE DEPTNO=V1;
BEGIN
OPEN C1(20);
LOOP
FETCH C1 INTO VV;
EXIT WHEN C1%NOTFOUND;
DBMS_OUTPUT.PUT_LINE(VV.ENAME||' '||VV.EMPNO||' '||VV.DEPTNO||' '||VV.SAL);
END LOOP;
CLOSE C1;
END;
/
