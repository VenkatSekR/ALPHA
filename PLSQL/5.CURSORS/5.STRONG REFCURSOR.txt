DECLARE 
TYPE RECDATA IS RECORD(MY_SAL EMP.SAL%TYPE);
TYPE C1_SREF IS REF CURSOR RETURN RECDATA;
MRC1SREF C1_SREF;
REC_VAR EMP.SAL%TYPE;
BEGIN
OPEN MRC1SREF FOR SELECT SAL FROM EMP WHERE EMPNO=7900;
FETCH MRC1SREF INTO REC_VAR;
DBMS_OUTPUT.PUT_LINE('EMPLOYEE SAL'||'='||REC_VAR);
CLOSE MRC1SREF;
END;
