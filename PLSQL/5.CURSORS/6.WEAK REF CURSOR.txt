DECLARE 
TYPE MRC1_SREF IS REF CURSOR;
C1_SREF MRC1_SREF; 
E_ENAME EMP.ENAME%TYPE;
ESAL    EMP.SAL%TYPE;
BEGIN
OPEN C1_SREF FOR SELECT ENAME,SAL FROM EMP WHERE EMPNO=7900;
FETCH C1_SREF INTO E_ENAME, ESAL;
CLOSE C1_SREF;
DBMS_OUTPUT.PUT_LINE(E_ENAME||' SALARY IS '||ESAL);
END;
/
