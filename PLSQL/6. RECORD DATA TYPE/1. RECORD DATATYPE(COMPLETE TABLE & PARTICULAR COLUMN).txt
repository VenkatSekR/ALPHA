DEFINITION :
RECORD IS A GROUP OF DATATYPE STORED IN MULTIPLE FIELDS WITH ITS OWN NAME AND HOLDING DIFFRENT DATA TYPES.

TYPES;
1.TABLE BASED RECORD.
2.CURSOR BASED RECORD.
3.USER DEFINED RECORD.

SYNTAX :

VARIABLE TABLE_NAME%ROWTYPE;
VARIABLE CURSOR_NAME%ROWTYPE;

TO COPY DATATYPES OF A COMPLETE TABLE :

SET SERVEROUTPUT ON;
DECLARE
  vv emp%ROWTYPE;
BEGIN
  SELECT * INTO VV FROM emp WHERE EMPNO = 7902;
  DBMS_OUTPUT.PUT_LINE (vv.ENAME ||' '||vv.sal);
  DBMS_OUTPUT.PUT_LINE (vv.HIREDATE);
END;

TO COPY DATATYPES OF A SINGLE COLUMN :

DECLARE
  vv emp%ROWTYPE;
BEGIN
  SELECT EMPNO, ENAME, SAL INTO VV.EMPNO, VV.ENAME, VV.SAL FROM emp WHERE EMPNO = 7902;
  DBMS_OUTPUT.PUT_LINE (vv.ENAME ||' '||vv.sal);
  DBMS_OUTPUT.PUT_LINE (vv.EMPNO);
END;
