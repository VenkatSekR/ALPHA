SYNTAX

DECLARE
  CURSOR C1 IS SELECT ENAME, SAL, JOB FROM EMP WHERE SAL<=2000;
  RDTN C1%ROWTYPE; --RECORD DATA TYPE NAME CURSORNAME%ROWTYPE
BEGIN
OPEN C1;
  FETCH C1 INTO RDTN; -- FETCH CURSOR NAME INTO RECORD DATA TYPE NAME;
  EXIT WHEN C1%NOTFOUND;
  DBMS_OUTPUT.PUT_LINE (RDTN.ENAME ||' '||RDTN.SAL); -- RECORD DATA TYPE NAME.COLUMN_NAME
  DBMS_OUTPUT.PUT_LINE (RDTN.JOB);
CLOSE C1;
END;
/
