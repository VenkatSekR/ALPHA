
DECLARE
TYPE UDDT IS RECORD(      -- TYPE (TYPE NAME)
VV_ENAME EMP.ENAME%TYPE,     VARIABLE TABLENAME.COLUMNNAME%TYPE 
VV_DEPTNO DEPT.DEPTNO%TYPE
);
RDTN UDDT; --(RECORD DATA NAME) (TYPE NAME) 
BEGIN
SELECT ENAME , DEPTNO INTO RDTN.VV_ENAME, RDTN.VV_DEPTNO FROM EMP JOIN DEPT USING (DEPTNO) WHERE EMPNO=7902;
DBMS_OUTPUT.PUT_LINE(RDTN.VV_ENAME||' '||RDTN.VV_DEPTNO); --(RECORD DATA NAME.VARIABLE NAME)
END;
/
