
CREATE OR REPLACE PROCEDURE SALARY_RAISE(SAL_RAISE NUMBER, DEPT_NO NUMBER) IS
BEGIN
UPDATE
EMP SET SAL = SAL*SAL_RAISE WHERE DEPTNO = DEPT_NO;
END;
/

EXEC SAL_RAISE(10,20);

CREATE OR REPLACE PROCEDURE SALARY_RAISE(SAL_RAISE NUMBER,DEPT_NO NUMBER) IS
