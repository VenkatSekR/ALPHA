CREATE OR REPLACE PACKAGE PKG IS 
FUNCTION F1 RETURN VARCHAR2;
PROCEDURE P1 (ENAME VARCHAR2,EMPNO NUMBER);
END PKG;
/


CREATE OR REPLACE PACKAGE BODY PKG IS
--FUNCTION
FUNCTION F1 RETURN VARCHAR2 IS 
BEGIN
RETURN 'VENKATSEKAR';
END F1;
--PROCEDURE
PROCEDURE P1(ENAME VARCHAR2, EMPNO NUMBER) IS 
BEGIN
INSERT INTO NAME_NO(ENAME,EMPNO) VALUES (ENAME, EMPNO);
END;
END PKG;
/


BEGIN
DBMS_OUTPUT.PUT_LINE(PKG.F1);
END;
/

SELECT * FROM NAME_NO;
/
